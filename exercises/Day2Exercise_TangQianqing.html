report_dataframe <- function(df) {
  cols <- colnames(df)
  for (i in cols) {
    print(paste0(i, ": ", mode(df[, i])))
  }
}

report_vector <- function(v, getmedian) {
  print(paste("Mean:", mean(v)))
  print(paste("Standard Deviation:", sd(v)))
  if (getmedian) {
    print(paste("Median:", median(v)))
  }
}
